class Solution {
  public:
    int majorityElement(vector<int> &nums) {
        int candidate = 0;
        int count = 0;
        for (int num : nums) {
            if (count == 0) {
                candidate = num;
            }
            if (candidate == num) {
                count += 1;
            } else {
                count -= 1;
            }
        }
        return candidate;
    }
};

class Solution {
  public:
    void sortColors(vector<int> &nums) {
        int redPos = 0;
        int bluePos = nums.size() - 1;
        int tmp = 0;
        for (int i = 0; i <= bluePos; ++i) {
            while (i <= bluePos && nums[i] == 2) {
                tmp = nums[bluePos];
                nums[bluePos] = 2;
                nums[i] = tmp;
                bluePos -= 1;
            }
            if (nums[i] == 0) {
                tmp = nums[redPos];
                nums[redPos] = 0;
                nums[i] = tmp;
                redPos += 1;
            }
        }
    }
};

class Solution {
  public:
    void nextPermutation(vector<int> &nums) {
        int n = nums.size();
        int i = n - 2;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            --i;
        }
        if (i >= 0) {
            for (int j = n - 1; j > i; --j) {
                if (nums[j] > nums[i]) {
                    int tmp = nums[i];
                    nums[i] = nums[j];
                    nums[j] = tmp;
                    break;
                }
            }
        }
        for (int left = i + 1, right = n - 1; left < right; ++left, --right) {
            int tmp = nums[left];
            nums[left] = nums[right];
            nums[right] = tmp;
        }
    }
};
